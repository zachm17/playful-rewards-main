<script>
// Timer functionality
function updateTimer() {
let minutes = 29;
let seconds = 30;
const timerElement = document.getElementById('prelander-timer');
setInterval(() => {
if (seconds === 0) {
if (minutes === 0) {
timerElement.textContent = '00:00';
return;
}
minutes--;
seconds = 59;
} else {
seconds--;
}
const displayMinutes = minutes < 10 ? '0' + minutes : minutes;
const displaySeconds = seconds < 10 ? '0' + seconds : seconds;
timerElement.textContent = `${displayMinutes}:${displaySeconds}`;
}, 1000);
}
// Notification functionality
const names = [
"James", "Emma", "Michael", "Sophia", "William", "Olivia", "John", "Ava",
"David", "Isabella", "Richard", "Mia", "Joseph", "Charlotte", "Thomas"
];
function generateAmount() {
return (Math.random() * (50 - 10) + 10).toFixed(2);
}
function generateTime() {
const times = ["just now", "1 minute ago", "2 minutes ago", "3 minutes ago",
"5 minutes ago"];
return times[Math.floor(Math.random() * times.length)];
}
function updateNotification() {
const notification = document.getElementById('notification');
notification.style.opacity = '0';
setTimeout(() => {
document.getElementById('notif-name').textContent =
names[Math.floor(Math.random() * names.length)];
document.getElementById('notif-amount').textContent =
generateAmount();
document.getElementById('notif-time').textContent = generateTime();
notification.style.opacity = '1';
}, 500);
}
function startCountdown() {
let minutes = 10;
let seconds = 0;
const countdownElement = document.getElementById('countdown');
const timer = setInterval(() => {
if (seconds === 0) {
if (minutes === 0) {
clearInterval(timer);
return;
}
minutes--;
seconds = 59;
} else {
seconds--;
}
countdownElement.textContent =
`${minutes}:${seconds.toString().padStart(2, '0')}`;
}, 1000);
}
function showMainLander() {
document.querySelector('.prelander').style.display = 'none';
document.querySelector('.mainlander').style.display = 'block';
window.scrollTo(0, 0);
updateNotification();
setInterval(updateNotification, 4000);
startCountdown();
}
// Start timer when page loads
window.onload = updateTimer;
</script>
